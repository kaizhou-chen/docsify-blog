## 节流



- 懒加载、滚动加载，或者监听滚动条位置

- 防止高频点击，防止表单重复提交

```js
/* 防抖 */
function throttle(fn, delay) {
	let valid = true;
	return function () {
		/* 一开始就触发---后续每隔delay时间内只触发一次 */
		if (valid) {
			valid = false; // 关闭阀门
			// 如果阀门已经打开，就继续往下
			setTimeout(() => {
				fn.apply(this, arguments); // 定时器结束后执行
				valid = true; // 执行完成后打开阀门
			}, delay);
		}
	};
}
```
